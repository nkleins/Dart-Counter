# Spiel-Engine (Fundament + x01) — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Eine reine, ohne I/O testbare TypeScript-Bibliothek, die aus einer flachen
Dart-Wurfliste den kompletten x01-Spielzustand berechnet (Punkte, Züge, Bust, In/Out,
Sieg). Undo entsteht dadurch gratis: einfach das letzte Event weglassen und neu rechnen.

**Architecture:** Der Zustand ist eine **reine Funktion** der geordneten Wurfliste.
Kein veränderlicher Zustand, keine Datenbank, keine Zeit — nur `computeX01State(options,
players, darts) → X01State`. Züge werden aus aufeinanderfolgenden Würfen desselben
Spielers gebildet (Spieler wechseln sich ab, daher = ein Zug), gedeckelt auf 3 Darts.

**Tech Stack:** Node ≥ 20, TypeScript (strict), Vitest als Test-Runner. Reines ESM.

## Global Constraints

- Sprache: **TypeScript strict** (`"strict": true`), keine `any` außer explizit begründet.
- Modul-System: **ESM** (`"type": "module"`), Datei-Importe mit `.js`-Endung im TS-Quelltext.
- Node-Version-Floor: **20**.
- Ein Dart wird einheitlich dargestellt als `{ segment, multiplier }`:
  - Miss: `segment: 0, multiplier: 0`
  - Zahl 1–20: `segment: 1..20`, `multiplier: 1|2|3` (Single/Double/Triple)
  - Bull außen (25): `segment: 25, multiplier: 1`
  - Bull innen (50): `segment: 25, multiplier: 2`  (zählt als **Double**)
  - `segment: 25, multiplier: 3` ist ungültig (kein Triple-Bull).
- Punkte eines Darts: `0` bei Miss, sonst `segment * multiplier`.
- x01-Startwerte: nur `301 | 501 | 701`. In/Out je `straight | double | master`
  (Master = Double **oder** Triple zählt zum Öffnen/Beenden).
- Alle Geldbeträge/Zeiten/Deadlines: hier keine (reine Logik).
- Commits: **nur Meilensteine, Autor nur der User, keine Co-Authored-By-Zeile.**
  (In diesem Plan = ein Commit pro Task ist ok, da jede Task ein abgeschlossener
  Engine-Baustein mit grünen Tests ist.)

`server/src/engine/x01.ts`:
```ts
import type { PlayerDart, PlayerInput, X01Options } from './types.js';
import { dartPoints } from './score.js';

export interface X01PlayerState {
  playerId: string;
  remaining: number;
  dartsThrown: number;
  finished: boolean;
}

export interface X01State {
  currentPlayerId: string | null;
  round: number;
  dartsThrownThisTurn: number;
  finished: boolean;
  winnerId: string | null;
  players: X01PlayerState[];
}

interface Turn {
  playerId: string;
  darts: PlayerDart[];
}

/** Aufeinanderfolgende Würfe desselben Spielers = ein Zug (max. 3 Darts). */
function groupTurns(darts: PlayerDart[]): Turn[] {
  const turns: Turn[] = [];
  for (const dart of darts) {
    const last = turns[turns.length - 1];
    if (last && last.playerId === dart.playerId && last.darts.length < 3) {
      last.darts.push(dart);
    } else {
      turns.push({ playerId: dart.playerId, darts: [dart] });
    }
  }
  return turns;
}

export function computeX01State(
  options: X01Options,
  players: PlayerInput[],
  darts: PlayerDart[],
): X01State {
  const order = [...players].sort((a, b) => a.order - b.order).map((p) => p.id);
  const ps = new Map<string, X01PlayerState>(
    players.map((p) => [p.id, { playerId: p.id, remaining: options.start, dartsThrown: 0, finished: false }]),
  );

  const turns = groupTurns(darts);
  let winnerId: string | null = null;
  let lastTurnComplete = true;

  for (let i = 0; i < turns.length; i++) {
    const turn = turns[i]!;
    const state = ps.get(turn.playerId);
    if (!state || winnerId) continue;
    for (const dart of turn.darts) {
      state.dartsThrown += 1;
      state.remaining -= dartPoints(dart);
    }
    if (i === turns.length - 1) lastTurnComplete = turn.darts.length >= 3;
  }

  return assembleState(order, ps, turns, winnerId, lastTurnComplete);
}

/**
 * Baut aus den Spielerzuständen + der Info, ob der letzte Zug abgeschlossen ist, den
 * öffentlichen X01State (aktueller Spieler, Runde, Darts im Zug). Wird von allen
 * Ausbaustufen von computeX01State genutzt und ändert sich ab hier nicht mehr — nur
 * die Replay-Schleife oben wächst in Task 4/5.
 */
function assembleState(
  order: string[],
  ps: Map<string, X01PlayerState>,
  turns: Turn[],
  winnerId: string | null,
  lastTurnComplete: boolean,
): X01State {
  const players_out = order.map((id) => ps.get(id)!);
  if (winnerId) {
    return {
      currentPlayerId: null,
      round: Math.max(1, Math.ceil(turns.length / Math.max(1, order.length))),
      dartsThrownThisTurn: 0,
      finished: true,
      winnerId,
      players: players_out,
    };
  }
  if (turns.length === 0) {
    return { currentPlayerId: order[0] ?? null, round: 1, dartsThrownThisTurn: 0, finished: false, winnerId: null, players: players_out };
  }
  const last = turns[turns.length - 1]!;
  const lastIdx = order.indexOf(last.playerId);
  const curIdx = lastTurnComplete ? (lastIdx + 1) % order.length : lastIdx;
  const currentPlayerId = order[curIdx]!;
  const dartsThrownThisTurn = lastTurnComplete ? 0 : last.darts.length;
  return {
    currentPlayerId,
    round: roundOf(turns, order, currentPlayerId, lastTurnComplete),
    dartsThrownThisTurn,
    finished: false,
    winnerId: null,
    players: players_out,
  };
}

/** Runde des aktuellen Spielers = Anzahl bereits abgeschlossener Züge dieses Spielers + 1. */
function roundOf(turns: Turn[], order: string[], currentPlayerId: string, lastComplete: boolean): number {
  let completed = 0;
  for (let i = 0; i < turns.length; i++) {
    const t = turns[i]!;
    const isLast = i === turns.length - 1;
    const complete = !isLast || lastComplete;
    if (t.playerId === currentPlayerId && complete) completed += 1;
  }
  return completed + 1;
}
```

    const turn = turns[i]!;
    const state = ps.get(turn.playerId);
    if (!state || winnerId) continue;

    const startRemaining = state.remaining;
    const startOpened = state.opened;
    let remaining = state.remaining;
    let opened = state.opened;
    let busted = false;
    let wonThisTurn = false;

    for (const dart of turn.darts) {
      state.dartsThrown += 1;

      if (!opened) {
        const opens = options.in === 'double' ? isDouble(dart) : isDoubleOrTriple(dart);
        if (!opens) continue; // Dart zählt nicht bis eröffnet
        opened = true;
      }

      const next = remaining - dartPoints(dart);
      if (next < 0) { busted = true; break; }
      if (next === 0) {
        const outOk =
          options.out === 'straight' ? true :
          options.out === 'double' ? isDouble(dart) :
          isDoubleOrTriple(dart); // master
        if (!outOk) { busted = true; break; }
        remaining = 0;
        wonThisTurn = true;
        break;
      }
      if (options.out === 'double' && next === 1) { busted = true; break; }
      remaining = next;
    }

    if (busted) {
      state.remaining = startRemaining;
      state.opened = startOpened || opened; // einmal eröffnet bleibt eröffnet
    } else {
      state.remaining = remaining;
      state.opened = opened;
      if (wonThisTurn) { state.finished = true; winnerId = turn.playerId; }
    }
    if (i === turns.length - 1) lastTurnComplete = busted || wonThisTurn || turn.darts.length >= 3;
  }
